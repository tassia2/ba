\section{Naming Conventions}
\subsection{Classes, functions and variables}
\begin{enumerate}
\item class names: \code{ThisIsAClassName}
\item function names: \code{this\_is\_a\_function\_name}
\item member variables get a underscore: \code{i\_am\_a\_member\_variable\_}
\item integer variables and functions returning a number of something: \code{nb\_XY}
\end{enumerate}
